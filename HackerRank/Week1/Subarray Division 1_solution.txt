#include<stdio.h>
int main()
{
    int n;
    scanf("%d",&n);
    int arr[n];
     for(int i=0;i<n;i++)
     {
        scanf("%d",&arr[i]);
     }
     
     int d,m;
     scanf("%d %d",&d,&m);
     int count = 0;    
     for(int i=0;i<=n-m;i++)
     {
        int sum =0;
        for(int j=0;j<m;j++)
        {
            sum = sum + arr[i+j];
        }
        if(sum==d)
        {
            count = count + 1;
        }
     }
     printf("%d",count);
     return 0;
}
