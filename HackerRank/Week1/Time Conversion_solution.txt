#include<stdio.h>
int main()
{
    char time[11];
    int hour;
    char minute[3], second[3], format[3];
    scanf("%s",time);
    sscanf(time,"%2d:%2s:%2s%2s",&hour,minute,second,format);
    
    if(format[0]=='A')
    {
        if(hour==12)
        {
            hour = 00;
        }
    }
    else 
    {
        if(format[0]=='P')
        {
            if(hour!=12)
            {
                hour = hour + 12;
            }
        }
    }
    printf("%02d:%s:%s\n",hour,minute,second);
    return 0;
}
