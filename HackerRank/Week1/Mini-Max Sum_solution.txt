#include<stdio.h>
int main()
{
    long long int arr[5];
    long long int total = 0;
    long long min, max;
    for(int i=0;i<5;i++)
    {
        scanf("%lld",&arr[i]);
        total = total + arr[i];
    }
    min = max = arr[0];
    for(int i=0;i<5;i++)
    {
        if(arr[i]>max)
        {
            max = arr[i];
        }
        if(arr[i]<min)
        {
            min = arr[i];
        }
    }
    long long int min_sum = total - max;
    long long int max_sum = total - min;
    printf("%lld %lld",min_sum,max_sum);
}
