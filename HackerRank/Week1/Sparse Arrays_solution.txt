#include<stdio.h>
#include<string.h>
int main()
{
    int n,q;
    scanf("%d",&n);
    char strings[1000][100];
    for(int i=0;i<n;i++)
    {
        scanf("%s",strings[i]);
    }
    scanf("%d",&q);
    char queries[1000][100];
    for(int i=0;i<q;i++)
    {
        scanf("%s",queries[i]);
    }
    for(int i=0;i<q;i++)
    {
        int count = 0;
        for(int j=0;j<n;j++)
        {
            if(strcmp(strings[j],queries[i])==0)
            {
                count = count + 1;
            }
        }
        printf("%d\n",count);
    }
    return 0;
}
